
//

AT+CWMODE=3 // softAP + station (модуль может работать и как AP, и как станция)

AT+CWJAP="MyWiFi)","11111111" // подключение к wifi

AT+CWLAP // Выводит список всех доступных точек доступа

AT+CWQAP // отключение от точки доступа

AT+CWLIF // Выводит список IP-адресов станций, подключенных к модулю, находящемуся в режиме softAP (точка доступа)

AT+CIPSTA? // Запрашивает информацию о IP-адресе станции ESP8266 (модуль работает как станция)
AT+CIPSTA=ip // Устанавливает значение IP-адреса станции ESP8266

AT+CIPAP? // Запрашивает информацию о IP-адресе точки доступа ESP8266 (модуль работает в режиме softAP)
AT+CIPAP=ip // Устанавливает значение IP-адреса точки доступа



AT+CIPMUX=0 // Устанавливает режим соединения: разрешает одиночное соединение
AT+CIPMUX=1 // Устанавливает режим соединения: разрешает множественные соединения

AT+CIPSERVER=1 // Конфигурирует и запускает сервер (по умолчанию используется порт 333)
AT+CIPSERVER=0 // удаляет сервер (после этого нужна перезагрузка)
